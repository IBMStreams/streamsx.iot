005package com.ibm.streamsx.iotf.sample;
006
007import java.io.File;
008import java.util.HashMap;
009import java.util.Map;
010
011import com.ibm.streamsx.iotf.DeviceEvent;
012import com.ibm.streamsx.iotf.IotfStreams;
013import com.ibm.streamsx.topology.TStream;
014import com.ibm.streamsx.topology.Topology;
015import com.ibm.streamsx.topology.TopologyElement;
016import com.ibm.streamsx.topology.context.AnalyticsServiceProperties;
017import com.ibm.streamsx.topology.context.StreamsContext.Type;
018import com.ibm.streamsx.topology.context.StreamsContextFactory;
019
020/**
021 * Sample application subscribing to device events.
022 * 
023 * @see com.ibm.streamsx.iotf.IotfStreams#eventsSubscribe(TopologyElement, String...)
024 */
025public class Events {
026
027    private Events() {}
028    
029    /**
030     * Execute this application against a Bluemix Streaming Analytic Service.
031     * 
032     * Usage:
033     * <BR>
034     * {@code java com.ibm.streamsx.iotf.sample.Events vcapFile serviceName [eventId ...]}
035     * @param args Arguments to the application.
036     * @throws Exception Exception executing 
037     */
038    public static void main(String[] args) throws Exception {
039        
040       
041        String vcapFile = args[0];
042        String serviceName = args[1];
043        String[] eventIds = null;
044        if (args.length > 2) {
045            eventIds = new String[args.length - 2];
046            System.arraycopy(args, 2, eventIds, 0, args.length - 2);
047        }
048        
049        // The declared application.
050        Topology topology = new Topology();
051        
052        // Subscribe to device events and just print them to standard out.
053        TStream<DeviceEvent> events = IotfStreams.eventsSubscribe(topology, eventIds);        
054        events.print();
055        
056        // Submit to BlueMix
057        Map<String,Object> config = new HashMap<>();      
058        config.put(AnalyticsServiceProperties.VCAP_SERVICES, new File(vcapFile));
059        config.put(AnalyticsServiceProperties.SERVICE_NAME, serviceName);
060        StreamsContextFactory.getStreamsContext(Type.ANALYTICS_SERVICE).submit(topology, config);
061    }
062}
