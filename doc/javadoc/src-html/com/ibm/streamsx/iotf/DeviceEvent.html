005package com.ibm.streamsx.iotf;
006
007import java.io.IOException;
008import java.io.Serializable;
009
010import com.ibm.json.java.JSON;
011import com.ibm.json.java.JSONObject;
012import com.ibm.streams.operator.Tuple;
013import com.ibm.streamsx.topology.tuple.JSONAble;
014
015/**
016 * A device event.
017 */
018public class DeviceEvent implements JSONAble, Serializable {
019    public static final String EVENT_ID = "eventId";
020    
021    private static final long serialVersionUID = 1L;
022    
023    private final Device device;
024    private final String eventId;
025    private final String ts;
026    private final JSONObject payload;
027    
028    public DeviceEvent(Device device, String eventId, JSONObject data) {
029        super();
030        this.device = device;
031        this.eventId = eventId;
032        this.ts = (String) data.get("ts");
033        this.payload = (JSONObject) data.get("d");
034        
035    }
036    public static DeviceEvent newDeviceEvent(Tuple tuple) {
037        return newDeviceEvent(
038                Device.newDevice(tuple),
039                tuple.getString(EVENT_ID),
040                tuple.getString(Device.JSON));
041    }
042    public static DeviceEvent newDeviceEvent(Device device, String eventId, String serializedData) {      
043        JSONObject payload;
044        try {
045            payload = (JSONObject) JSON.parse(serializedData);
046        } catch (IOException e) {
047            throw new RuntimeException(e);
048        }        
049        return new DeviceEvent(device, eventId, payload);       
050    }
051
052    public Device getDevice() {
053        return device;
054    }
055
056    public String getEventId() {
057        return eventId;
058    }
059
060    public JSONObject getPayload() {
061        return payload;
062    }
063    public String getTs() {
064        return ts;
065    }
066    
067    @Override
068    public JSONObject toJSON() {
069        JSONObject json = getDevice().toJSON();
070        json.put(EVENT_ID, getEventId());
071        json.put("ts", getTs());
072        json.put("payload", getPayload());
073        return json;
074    }
075    @Override
076    public String toString() {
077        try {
078            return toJSON().serialize();
079        } catch (IOException e) {
080            throw new RuntimeException(e);
081        }
082    }
083    
084    
085    
086}
