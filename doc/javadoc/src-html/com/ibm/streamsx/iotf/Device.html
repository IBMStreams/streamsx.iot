005package com.ibm.streamsx.iotf;
006
007import java.io.IOException;
008import java.io.Serializable;
009
010import com.ibm.json.java.JSONObject;
011import com.ibm.streams.operator.Tuple;
012import com.ibm.streamsx.topology.spl.SPLSchemas;
013import com.ibm.streamsx.topology.tuple.JSONAble;
014
015/**
016 * A device.
017 *
018 */
019public class Device implements JSONAble, Serializable {
020    
021    public static final String TYPE_ID = "typeId";
022    public static final String DEVICE_ID = "deviceId";
023    public static final String JSON = SPLSchemas.JSON.getAttribute(0).getName();
024    
025    
026    private static final long serialVersionUID = 1L;
027    
028    private final String typeId;
029    private final String id;
030    
031    public Device(String typeId, String id) {
032        super();
033        this.typeId = typeId;
034        this.id = id;
035    }
036    
037    public static Device newDevice(Tuple tuple) {
038        return new Device(
039                tuple.getString(Device.TYPE_ID),
040                tuple.getString(Device.DEVICE_ID));
041    }
042
043    
044    /**
045     * Get the device type identifier.
046     * @return device type identifier.
047     */
048    public String getTypeId() {
049        return typeId;
050    }
051
052    /**
053     * Get the device identifier.
054     * @return device identifier.
055     */
056    public String getId() {
057        return id;
058    }
059    
060    @Override
061    public JSONObject toJSON() {
062        JSONObject json = new JSONObject();       
063        json.put(TYPE_ID, getTypeId());
064        json.put(DEVICE_ID, getId());       
065        return json;
066    }
067    
068    @Override
069    public String toString() {
070        try {
071            return toJSON().serialize();
072        } catch (IOException e) {
073            throw new RuntimeException(e);
074        }
075    }
076
077    @Override
078    public int hashCode() {
079        final int prime = 31;
080        int result = 1;
081        result = prime * result + ((id == null) ? 0 : id.hashCode());
082        result = prime * result + ((typeId == null) ? 0 : typeId.hashCode());
083        return result;
084    }
085
086    @Override
087    public boolean equals(Object obj) {
088        if (this == obj)
089            return true;
090        if (obj == null)
091            return false;
092        if (getClass() != obj.getClass())
093            return false;
094        Device other = (Device) obj;
095        if (id == null) {
096            if (other.id != null)
097                return false;
098        } else if (!id.equals(other.id))
099            return false;
100        if (typeId == null) {
101            if (other.typeId != null)
102                return false;
103        } else if (!typeId.equals(other.typeId))
104            return false;
105        return true;
106    }
107}
