005package com.ibm.streamsx.iot;
006
007import java.io.IOException;
008import java.io.Serializable;
009
010import com.ibm.json.java.JSON;
011import com.ibm.json.java.JSONObject;
012import com.ibm.streams.operator.OutputTuple;
013import com.ibm.streams.operator.Tuple;
014import com.ibm.streamsx.topology.tuple.JSONAble;
015
016/**
017 * A device command.
018 */
019public class DeviceCmd implements JSONAble, Serializable {
020    private static final long serialVersionUID = 1L;
021
022    public static final String CMD_ID = "cmdId";
023
024    private final Device device;
025    private final String cmdId;
026    private final JSONObject payload;
027
028    public DeviceCmd(Device device, String cmdId, JSONObject payload) {
029        super();
030        this.device = device;
031        this.cmdId = cmdId;
032        this.payload = payload;
033    }
034
035    public static DeviceCmd newDeviceCmd(Tuple tuple) {
036        return newDeviceCmd(Device.newDevice(tuple), tuple.getString(CMD_ID), tuple.getString(Device.JSON));
037    }
038
039    public static DeviceCmd newDeviceCmd(Device device, String cmdId, String serializedData) {
040        JSONObject payload;
041        try {
042            payload = (JSONObject) JSON.parse(serializedData);
043        } catch (IOException e) {
044            throw new RuntimeException(e);
045        }
046        return new DeviceCmd(device, cmdId, payload);
047    }
048
049    public OutputTuple toTuple(OutputTuple tuple) {
050        tuple.setString(Device.TYPE_ID, getDevice().getTypeId());
051        tuple.setString(Device.DEVICE_ID, getDevice().getId());
052        tuple.setString(DeviceCmd.CMD_ID, getCmdId());
053        try {
054            tuple.setString(Device.JSON, getPayload().serialize());
055        } catch (Exception e) {
056            throw new RuntimeException(e);
057        }
058        return tuple;
059    }
060
061    public Device getDevice() {
062        return device;
063    }
064
065    public String getCmdId() {
066        return cmdId;
067    }
068
069    public JSONObject getPayload() {
070        return payload;
071    }
072
073    @Override
074    public JSONObject toJSON() {
075        JSONObject json = getDevice().toJSON();
076        json.put("cmdId", getCmdId());
077        json.put("payload", getPayload());
078        return json;
079    }
080
081    @Override
082    public String toString() {
083        try {
084            return toJSON().serialize();
085        } catch (IOException e) {
086            throw new RuntimeException(e);
087        }
088    }
089}
