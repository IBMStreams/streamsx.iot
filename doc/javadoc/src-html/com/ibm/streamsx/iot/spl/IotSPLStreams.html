005package com.ibm.streamsx.iot.spl;
006
007import java.util.ArrayList;
008import java.util.HashMap;
009import java.util.List;
010import java.util.Map;
011
012import com.ibm.streams.operator.Attribute;
013import com.ibm.streams.operator.Type;
014import com.ibm.streamsx.topology.TopologyElement;
015import com.ibm.streamsx.topology.spl.SPL;
016import com.ibm.streamsx.topology.spl.SPLStream;
017
018/**
019 * IBM Watson IoT Platform SPL streams.
020 * 
021 * These streams using the microservice model promoted by the
022 * {@code com.ibm.streamsx.iot} toolkit. In order to interact with IoT Platform
023 * the SPL application
024 * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be running.
025 *
026 */
027public class IotSPLStreams {
028
029    private IotSPLStreams() {
030    }
031
032    /**
033     * Subscribe to device events as SPL tuples.
034     * 
035     * Subscribes to all device events with event identifiers listed in
036     * {@code eventId}. If {@code eventId} is empty or passed as a {@code null}
037     * array then all device events are subscribed to.
038     * 
039     * To receive device events the SPL application
040     * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be
041     * running in the same Streams instance.
042     * 
043     * @param te
044     *            Topology to create this source in.
045     * @param eventId
046     *            Event identifiers to subscribe to. If no event identf
047     * @return Stream containing device events.
048     * 
049     * @see com.ibm.streamsx.iot.IotStreams#eventsSubscribe(TopologyElement,
050     *      String...)
051     */   
052    public static SPLStream eventsSubscribe(TopologyElement te, String[] typeIds, String... eventId) {
053
054        Map<String, Object> params = new HashMap<>();
055        if (typeIds != null && typeIds.length != 0)
056            params.put("typeIds", listRStringParam(typeIds));
057
058        if (eventId != null && eventId.length != 0)
059            params.put("eventIds", listRStringParam(eventId));    
060
061        return SPL.invokeSource(te, "com.ibm.streamsx.iot::EventsSubscribe", params, Schemas.DEVICE_EVENT);
062    }
063    
064    public static SPLStream statusesSubscribe(TopologyElement te, String ... typeId) {
065
066        Map<String, Object> params = new HashMap<>();
067        if (typeId != null && typeId.length != 0)
068            params.put("typeIds", listRStringParam(typeId));
069 
070        return SPL.invokeSource(te, "com.ibm.streamsx.iot::StatusesSubscribe", params, Schemas.DEVICE_STATUS);
071    }
072
073    /**
074     * Subscribe to sent device commands as SPL tuples.
075     * 
076     * Subscribes to all device commands with command identifiers listed in
077     * {@code cmdId}. If {@code cmdId} is empty or passed as a {@code null}
078     * array then all device commands are subscribed to.
079     * 
080     * To receive device events the SPL application
081     * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be
082     * running in the same Streams instance.
083     * 
084     * @param te
085     *            Topology to create this source in.
086     * @param cmdId
087     *            Command identifiers to subscribe to. If no command identifier
088     *            is passed then all commands are subscribed to.
089     * @return Stream containing device events.
090     * 
091     * @see com.ibm.streamsx.iot.IotStreams#eventsSubscribe(TopologyElement,
092     *      String...)
093     */
094    public static SPLStream commandsSubscribe(TopologyElement te, String[] typeIds, String... cmdId) {
095
096        Map<String, Object> params = new HashMap<>();
097        if (typeIds != null && typeIds.length != 0)
098            params.put("typeIds", listRStringParam(typeIds));
099
100        if (cmdId != null && cmdId.length != 0)
101            params.put("cmdIds", listRStringParam(cmdId));
102
103        return SPL.invokeSource(te, "com.ibm.streamsx.iot::CommandsSubscribe", params, Schemas.DEVICE_CMD);
104    }
105
106    private static Object listRStringParam(String[] values) {
107        List<String> literals = new ArrayList<>(values.length);
108        for (String v : values)
109            literals.add("\"" + v + "\"");
110
111        return new Attribute() {
112
113            @Override
114            public int getIndex() {
115                // TODO Auto-generated method stub
116                return 0;
117            }
118
119            @Override
120            public String getName() {
121                return literals.toString();
122            }
123
124            @Override
125            public Type getType() {
126                // TODO Auto-generated method stub
127                return null;
128            }
129
130            @Override
131            public boolean same(Attribute arg0) {
132                return false;
133            }
134
135        };
136    }
137
138    public static void commandPublish(SPLStream commandStream) {
139        if (!commandStream.getSchema().equals(Schemas.DEVICE_CMD))
140            throw new IllegalArgumentException("Schema is invalid:" + commandStream.getSchema().getLanguageType());
141        SPL.invokeSink("com.ibm.streamsx.iot::CommandPublish", commandStream, null);
142    }
143}
