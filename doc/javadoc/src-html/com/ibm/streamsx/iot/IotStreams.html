005package com.ibm.streamsx.iot;
006
007import com.ibm.streamsx.datetime.convert.ISO8601;
008import com.ibm.streamsx.iot.spl.IotSPLStreams;
009import com.ibm.streamsx.iot.spl.Schemas;
010import com.ibm.streamsx.topology.TStream;
011import com.ibm.streamsx.topology.TopologyElement;
012import com.ibm.streamsx.topology.spl.SPLStream;
013import com.ibm.streamsx.topology.spl.SPLStreams;
014
015/**
016 * IBM Watson IoT Platform streams.
017 * 
018 * These streams using the micro-service model promoted by the
019 * {@code com.ibm.streamsx.iot} toolkit. In order to interact with IoT Platform
020 * the Streams application
021 * {@code com.ibm.streamsx.iot.watson.apps::IotPlatform} must be running.
022 *
023 */
024public class IotStreams {
025
026    private IotStreams() {
027    }
028
029    /**
030     * Subscribe to events as {@code DeviceEvent} instances.
031     * 
032     * Subscribes to all device events with event identifiers listed in
033     * {@code eventId}. If {@code eventId} is empty or passed as a {@code null}
034     * array then all device events are subscribed to.
035     * 
036     * To receive device events the Streams application
037     * {@code com.ibm.streamsx.iot.watson.apps::IotPlatform} must be
038     * running in the same Streams instance.
039     * 
040     * @param te
041     *            Topology to create this source in.
042     * @param eventId
043     *            Event identifiers to subscribe to. If none are provided then
044     *            all event identifiers are subscribed to.
045     * @return Stream containing device events.
046     * 
047     */
048    public static TStream<DeviceEvent> eventsSubscribe(TopologyElement te, String... eventId) {
049        
050        return eventsSubscribe(te, null, eventId);
051    }
052    
053    /**
054     * Subscribe to events as {@code DeviceEvent} instances.
055     * 
056     * Subscribes to all device events with:
057     * <UL>
058     * <LI>All device type identifiers listed in {@code typeIds}. If
059     * {@code typeIds} is empty or passed as a {@code null} array reference
060     * then no filtering against type identifier is applied.</LI>
061     * <LI>AND</LI>
062     * <LI>All device event identifiers listed in {@code eventId}. If
063     * {@code eventId} is empty or passed as a {@code null} array reference
064     * then no filtering against event identifier is applied.</LI>
065     * </UL>
066     * 
067     * 
068     * To receive device events the Streams application
069     * {@code com.ibm.streamsx.iot.watson.apps::IotPlatform} must be
070     * running in the same Streams instance.
071     * 
072     * @param te
073     *            Topology to create this source in.
074     * @param typeIds Type identifers to subscribe to. If this is empty or null then
075     * all type identifiers are subscribed to.
076     * @param eventId
077     *            Event identifiers to subscribe to. If none are provided then
078     *            all event identifiers are subscribed to.
079     * @return Stream containing device events.
080     * 
081     */
082    public static TStream<DeviceEvent> eventsSubscribe(TopologyElement te, String[] typeIds, String... eventId) {
083        addDependentClasses(te);
084
085        SPLStream splEvents = IotSPLStreams.eventsSubscribe(te, typeIds, eventId);
086        TStream<DeviceEvent> events = splEvents.transform(IotUtils::newDeviceEvent);
087        return events.asType(DeviceEvent.class);
088    }    
089    
090
091    /**
092     * Subscribe to commands as {@code DeviceCmd} instances.
093     * 
094     * Subscribes to all device commands with commands identifiers listed in
095     * {@code cmdId}. If {@code cmdId} is empty or passed as a {@code null}
096     * array then all device commands are subscribed to.
097     * 
098     * To receive device events the Streams application
099     * {@code com.ibm.streamsx.iot.watson.apps::IotPlatform} must be
100     * running in the same Streams instance.
101     * 
102     * @param te
103     *            Topology to create this source in.
104     * @param cmdId
105     *            Command identifiers to subscribe to.
106     * @return Stream containing device events.
107     * 
108     */
109    public static TStream<DeviceCmd> commandsSubscribe(TopologyElement te, String... cmdId) {
110        return commandsSubscribe(te, null, cmdId);
111    }
112    
113    static void addDependentClasses(TopologyElement te) {
114        te.topology().addClassDependency(ISO8601.class);
115        te.topology().addClassDependency(DeviceEvent.class);
116    }
117    
118    public static TStream<DeviceCmd> commandsSubscribe(TopologyElement te, String[] typeIds, String... cmdId) {
119        addDependentClasses(te);
120        
121        SPLStream splCommands = IotSPLStreams.commandsSubscribe(te, typeIds, cmdId);
122        TStream<DeviceCmd> commands = splCommands.transform(IotUtils::newDeviceCmd);
123        return commands.asType(DeviceCmd.class);
124    }
125
126    public static void commandPublish(TStream<DeviceCmd> commandStream) {
127        addDependentClasses(commandStream);
128        
129        commandStream = commandStream.asType(DeviceCmd.class);
130        SPLStream splCommands = SPLStreams.convertStream(commandStream, IotUtils::toTuple,
131                Schemas.DEVICE_CMD);
132
133        IotSPLStreams.commandPublish(splCommands);
134    }
135
136    public static TStream<DeviceStatus> statusesSubscribe(TopologyElement te, String ... typeId) {
137        addDependentClasses(te);
138        
139        SPLStream splStatuses = IotSPLStreams.statusesSubscribe(te, typeId);
140        TStream<DeviceStatus> statuses = splStatuses.transform(DeviceStatus::newDeviceStatus);
141        return statuses.asType(DeviceStatus.class);
142    }
143}
