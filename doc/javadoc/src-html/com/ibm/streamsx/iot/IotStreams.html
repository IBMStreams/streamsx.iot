005package com.ibm.streamsx.iot;
006
007import com.ibm.streamsx.iot.spl.IotSPLStreams;
008import com.ibm.streamsx.iot.spl.Schemas;
009import com.ibm.streamsx.topology.TStream;
010import com.ibm.streamsx.topology.TopologyElement;
011import com.ibm.streamsx.topology.spl.SPLStream;
012import com.ibm.streamsx.topology.spl.SPLStreams;
013
014/**
015 * IBM Watson IoT Platform streams.
016 * 
017 * These streams using the micro-service model promoted by the
018 * {@code com.ibm.streamsx.iot} toolkit. In order to interact with IoT Platform
019 * the SPL application
020 * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be running.
021 *
022 */
023public class IotStreams {
024
025    private IotStreams() {
026    }
027
028    /**
029     * Subscribe to events as {@code DeviceEvent} instances.
030     * 
031     * Subscribes to all device events with event identifiers listed in
032     * {@code eventId}. If {@code eventId} is empty or passed as a {@code null}
033     * array then all device events are subscribed to.
034     * 
035     * To receive device events the SPL application
036     * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be
037     * running in the same Streams instance.
038     * 
039     * @param te
040     *            Topology to create this source in.
041     * @param eventId
042     *            Event identifiers to subscribe to.
043     * @return Stream containing device events.
044     * 
045     */
046    public static TStream<DeviceEvent> eventsSubscribe(TopologyElement te, String... eventId) {
047
048        SPLStream splEvents = IotSPLStreams.eventsSubscribe(te, eventId);
049        TStream<DeviceEvent> events = splEvents.transform(DeviceEvent::newDeviceEvent);
050        return events;
051    }
052
053    /**
054     * Subscribe to commands as {@code DeviceCmd} instances.
055     * 
056     * Subscribes to all device commands with commands identifiers listed in
057     * {@code cmdId}. If {@code cmdId} is empty or passed as a {@code null}
058     * array then all device commands are subscribed to.
059     * 
060     * To receive device events the SPL application
061     * {@code com.ibm.streamsx.iot.watson.apps::IotfOrganization} must be
062     * running in the same Streams instance.
063     * 
064     * @param te
065     *            Topology to create this source in.
066     * @param cmdId
067     *            Command identifiers to subscribe to.
068     * @return Stream containing device events.
069     * 
070     */
071    public static TStream<DeviceCmd> commandsSubscribe(TopologyElement te, String... cmdId) {
072
073        SPLStream splCommands = IotSPLStreams.commandsSubscribe(te, cmdId);
074        TStream<DeviceCmd> commands = splCommands.transform(DeviceCmd::newDeviceCmd);
075        return commands;
076    }
077
078    public static void commandPublish(TStream<DeviceCmd> commandStream) {
079        SPLStream splCommands = SPLStreams.convertStream(commandStream, (cmd, tuple) -> cmd.toTuple(tuple),
080                Schemas.DEVICE_CMD);
081
082        IotSPLStreams.commandPublish(splCommands);
083    }
084
085}
