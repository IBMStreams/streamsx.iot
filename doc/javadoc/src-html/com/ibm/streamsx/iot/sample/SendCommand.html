005package com.ibm.streamsx.iot.sample;
006
007import static java.util.concurrent.TimeUnit.SECONDS;
008
009import java.io.File;
010import java.util.Date;
011import java.util.HashMap;
012import java.util.Map;
013
014import com.ibm.json.java.JSONObject;
015import com.ibm.streamsx.iot.Device;
016import com.ibm.streamsx.iot.DeviceCmd;
017import com.ibm.streamsx.iot.IotStreams;
018import com.ibm.streamsx.topology.TStream;
019import com.ibm.streamsx.topology.Topology;
020import com.ibm.streamsx.topology.TopologyElement;
021import com.ibm.streamsx.topology.context.AnalyticsServiceProperties;
022import com.ibm.streamsx.topology.context.StreamsContext.Type;
023import com.ibm.streamsx.topology.context.StreamsContextFactory;
024
025/**
026 * Sample application subscribing to device commands.
027 * 
028 * @see com.ibm.streamsx.iot.IotStreams#commandsSubscribe(TopologyElement,
029 *      String...)
030 */
031public class SendCommand {
032
033    private SendCommand() {
034    }
035
036    /**
037     * Execute this application against a Bluemix Streaming Analytic Service.
038     * 
039     * Usage: <BR>
040     * {@code java com.ibm.streamsx.iot.sample.SendCommand vcapFile serviceName typeId deviceId message}
041     * 
042     * @param args
043     *            Arguments to the application.
044     * @throws Exception
045     *             Exception executing
046     */
047    public static void main(String[] args) throws Exception {
048
049        String vcapFile = args[0];
050        String serviceName = args[1];
051        String typeId = args[2];
052        String deviceId = args[3];
053        String message = args[4];
054
055        // The declared application.
056        Topology topology = new Topology();
057
058        final Device device = new Device(typeId, deviceId);
059
060        TStream<DeviceCmd> command = topology.periodicSource(() -> {
061            JSONObject payload = new JSONObject();
062            payload.put("msg", message + " @ " + new Date().toString());
063            return new DeviceCmd(device, "display", payload);
064        }, 10, SECONDS);
065
066        IotStreams.commandPublish(command);
067
068        // Submit to BlueMix
069        Map<String, Object> config = new HashMap<>();
070        config.put(AnalyticsServiceProperties.VCAP_SERVICES, new File(vcapFile));
071        config.put(AnalyticsServiceProperties.SERVICE_NAME, serviceName);
072        StreamsContextFactory.getStreamsContext(Type.ANALYTICS_SERVICE).submit(topology, config);
073    }
074
075}
