005package com.ibm.streamsx.iot.sample;
006
007import java.io.File;
008import java.util.HashMap;
009import java.util.Map;
010
011import com.ibm.streamsx.iot.DeviceEvent;
012import com.ibm.streamsx.iot.IotStreams;
013import com.ibm.streamsx.topology.TStream;
014import com.ibm.streamsx.topology.Topology;
015import com.ibm.streamsx.topology.TopologyElement;
016import com.ibm.streamsx.topology.context.AnalyticsServiceProperties;
017import com.ibm.streamsx.topology.context.StreamsContext.Type;
018import com.ibm.streamsx.topology.context.StreamsContextFactory;
019
020/**
021 * Sample application subscribing to device events.
022 * 
023 * @see com.ibm.streamsx.iot.IotStreams#eventsSubscribe(TopologyElement,
024 *      String...)
025 */
026public class Events {
027
028    private Events() {
029    }
030
031    /**
032     * Execute this application against a IBM Cloud Streaming Analytic Service.
033     * 
034     * Usage: <BR>
035     * {@code java com.ibm.streamsx.iot.sample.Events vcapFile serviceName [eventId ...]}
036     * 
037     * @param args
038     *            Arguments to the application.
039     * @throws Exception
040     *             Exception executing
041     */
042    public static void main(String[] args) throws Exception {
043
044        String vcapFile = args[0];
045        String serviceName = args[1];
046        String[] eventIds = null;
047        if (args.length > 2) {
048            eventIds = new String[args.length - 2];
049            System.arraycopy(args, 2, eventIds, 0, args.length - 2);
050        }
051
052        // The declared application.
053        Topology topology = new Topology();
054
055        // Subscribe to device events and just print them to standard out.
056        TStream<DeviceEvent> events = IotStreams.eventsSubscribe(topology, eventIds);
057        events.print();
058
059        // Submit to IBM Cloud
060        Map<String, Object> config = new HashMap<>();
061        config.put(AnalyticsServiceProperties.VCAP_SERVICES, new File(vcapFile));
062        config.put(AnalyticsServiceProperties.SERVICE_NAME, serviceName);
063        StreamsContextFactory.getStreamsContext(Type.ANALYTICS_SERVICE).submit(topology, config);
064    }
065}
